#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define MAX_SIZE 5
#define SWAP(x,y,t)((t)=(x),(x)=(y),(y)=(t))

typedef struct{
	int id;
	char name[20];
}element;

void selection_sort(element list[],int n)
{
	int i,j,least;
	element temp;
	for(i=0;i<n;i++){
		least=i;
		for(j=i+1; j<n;j++)
			if(list[least].id>list[j].id) least=j;
		SWAP(list[i],list[least],temp);
	}
}

void selection_sort2(element list[], int n)
{

	element temp;
	for(i=0;i<n;i++){
		least=i;
		for(j=i+1; j<n;j++)
			if(strcmp(list[least].name, list[j].name)>0) least=j;
		SWAP(list[i],list[least],temp);
	}
}
